# ManezCo
For Deployment
Backend:
1. cd djangoProject2
2. run "pip install -r requirements.txt"
3. run "python manage.py runserver"
Frontend:
1. cd frontend
2. run "npm install"
3. run "npm start"
Database:
1. use mysql
2. set mysql and create database 'manez'
run "create database manez charset utf8mb4;"
parameters:
'ENGINE': 'django.db.backends.mysql',
'NAME': 'manez',
'USER': 'root',
'PASSWORD': '123456',
'PORT': 3306,
'HOST': 'localhost',
